网格体系横表架构nmk.grid_basic
字段 字段名称 字段类型 字段描述
ID 网格编码 VARCHAR(32) 　
NAME 网格名称 VARCHAR(256) 　
PID 父节点ID VARCHAR(32) 　
TYPE 网格类型 VARCHAR(32) SPE:特征网格  SA：SA  ADMIN：物理网格
CITY_ID 地市编码 VARCHAR(32) 　
PARTITION_TYPE 网格分割方式 VARCHAR(32) 　
CHANNEL_CODE1 网格包含营业厅编码1 VARCHAR(32) 　
CHANNEL_ADDRESS1 网格包含营业厅地址1 VARCHAR(512) 　
CHANNEL_CODE2 网格包含营业厅编码2 VARCHAR(32) 　
CHANNEL_ADDRESS2 网格包含营业厅地址2 VARCHAR(512) 　
CHANNEL_CODE3 网格包含营业厅编码3 VARCHAR(32) 　
CHANNEL_ADDRESS3 网格包含营业厅地址3 VARCHAR(512) 　
STREET 街道 VARCHAR(512) 　
SECTION 区 VARCHAR(512) 　
ACREAGE 面积 DECIMAL(4, 0) 　
CHARGER_NAME 负责人姓名 VARCHAR(32) 　
CHARGER_PHONE 负责人号码 VARCHAR(32) 　
STAFF_ID 客户经理编码 VARCHAR(32) 　
STAFF_NAME 客户经理姓名 VARCHAR(32) 　
LEVEL 网格级别 DECIMAL(4, 0) 3 主网格 4 子网格 5 微网格
REMARK 备注 VARCHAR(512)
